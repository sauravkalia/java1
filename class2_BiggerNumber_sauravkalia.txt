
public class BiggerNumber {
	public static void main(String[] args)
	{
		int a = 10,b = 20,c =5;
		
			if(a>=b && a>=c) {
			
				System.out.println(a+" is bigger");
			}
			
			
			else if(b>=a && b>=c) {
				System.out.println(b+" is bigger");
			}
			else
			{
				System.out.println(c+" is bigger");
			}			
			
	}
}
